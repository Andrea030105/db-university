1. Selezionare tutti gli studenti iscritti al Corso di Laurea in Economia

    SELECT studente.* 
    FROM `studente`
    JOIN corso_di_laurea ON studente.corso_laurea_id = corso_di_laurea.id
    WHERE corso_di_laurea.nome LIKE "Economia Aziendale (Triennale)";

2. Selezionare tutti i Corsi di Laurea Magistrale del Dipartimento di Neuroscienze

    SELECT corso_di_laurea.*, dipartimenti.nome AS Nome_Dip
    FROM corso_di_laurea
    JOIN dipartimenti ON dipartimenti.id = corso_di_laurea.dipartimenti_id
    WHERE dipartimenti.nome LIKE "Neuroscienze" AND corso_di_laurea.nome LIKE "%magistrale%";

3. Selezionare tutti i corsi in cui insegna Fulvio Amato (id=44)

    SELECT corsi.*, insegnanti.nome AS Nome_Insegnante
    FROM corsi
    JOIN insegnanti ON corsi.id = insegnanti.corsi_id
    WHERE insegnanti.id = 44;

4. Selezionare tutti gli studenti con i dati relativi al corso di laurea a cui sono iscritti e il
relativo dipartimento, in ordine alfabetico per cognome e nome

    SELECT studente.*, corso_di_laurea.*, dipartimenti.*
    FROM studente
    JOIN corso_di_laurea ON corso_di_laurea.id = studente.corso_laurea_id
    JOIN dipartimenti ON dipartimenti.id = corso_di_laurea.dipartimenti_id
    ORDER BY studente.nome, studente.cognome ASC;

5. Selezionare tutti i corsi di laurea con i relativi corsi e insegnanti

    SELECT 
        corso_di_laurea.id   AS id_corso_laurea,
        corso_di_laurea.nome AS nome_corso_laurea,
        corsi.id             AS id_corso,
        corsi.nome           AS nome_corso,
        corsi.cfu,
        insegnanti.id        AS id_insegnante,
        insegnanti.nome      AS nome_insegnante,
        insegnanti.cognome   AS cognome_insegnante,
        insegnanti.livello
    FROM corso_di_laurea
    JOIN corsi      ON corsi.corso_laurea_id = corso_di_laurea.id
    JOIN insegnanti ON insegnanti.corsi_id   = corsi.id;
